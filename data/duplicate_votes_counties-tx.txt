SQL to Reproduce:

    SELECT
        dupes.county, COUNT(*),
        ROUND(c2016.republican * 100, 2) republican, ROUND(c2016.democrat * 100, 2) democrat
    FROM
    (
        SELECT
            county, last_name, given_names, v.voter_id,
            voting_method, precinct, recorded_on
        FROM voter_rolls_v2 v
        JOIN (
            SELECT voter_id, COUNT(*)
            FROM voter_rolls_v2
            WHERE voter_id != 0
            GROUP BY voter_id
            HAVING COUNT(*) > 1
        ) v2 ON v2.voter_id=v.voter_id
        ORDER BY county, v.last_name, v.given_names
    ) dupes
    JOIN counties_by_pol_party_2016_v2 c2016
        ON c2016.county=dupes.county
    GROUP BY dupes.county
    ORDER BY COUNT(*) DESC;
